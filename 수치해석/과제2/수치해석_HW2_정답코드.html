import numpy as np

def TDMA(a:np.array, b:np.array, c:np.array, r:np.array) -> tuple:
    n = r.shape[0]-1
    x = np.zeros(n+1)
    logs = []
    a = a.astype(np.float64)
    b = b.astype(np.float64)
    c = c.astype(np.float64)
    r = r.astype(np.float64)

    # loop for the delet of lower diagonal terms in the tridiagonal matrix
    for i in range(1, n+1):
        factor = a[i-1]/b[i-1]
        b[i] = b[i] - factor * c[i-1]
        r[i] = r[i] - factor * r[i-1]
        logs.append(f'Row {i+1} Forward Elimination: \na={a}\nb={b}\nc={c}\nr={r}')

    x[n]=r[n]/b[n]    
    # this loop is for backward substitution
    x[n] = r[n]/b[n]
    for i in range(n-1, -1, -1):
        x[i]=(r[i]-c[i]*x[i+1])/b[i]
    
    return logs, x

import numpy as np

def gauss_seidel(A:np.array, b:np.array, epsilon:float, max_iter:int) -> tuple:
    """
    # Gauss-Seidel Iteration Method for solving linear equations Ax = b
    
    ## Parameters:
        **A (2D numpy array)**: Coefficient matrix
        **b (1D numpy array)**: Right-hand side vector
        **epsilon (float)**: Convergence tolerance
        **max_iter (int)**: Maximum number of iterations
    
    Returns:
        **x** 1D numpy array, Solution vector\\
        **iterations (int)**: Number of iterations performed
    """
    n = len(b)
    x = np.zeros(n)  # Initial guess for solution
    x_old = np.zeros(n)
    A = A.astype(np.float64)
    b = b.astype(np.float64)
    
    for j in range(max_iter):
        sum_new = 0.0
        sum_old = 0.0
        
        for i in range(n):
            sum_ = 0.0
            
            # Compute the sum excluding the current variable x[i]
            for k in range(n):
                if i != k:
                    sum_ -= A[i][k] * x_old[k]
            
            # Update the current variable x[i]
            x[i] = (sum_ + b[i]) / A[i][i]
            
            # Update sums for convergence check
            sum_new += x[i]
            sum_old += x_old[i]
            
            # Update the old solution with the new value
            x_old[i] = x[i]
        
        # Check convergence
        res = abs(sum_new - sum_old)
        print(f"Iteration #{j+1}")
        print(f"Sum Old: {sum_old}")
        print(f"Sum New: {sum_new}")
        print(f"Residual: {res}")
        
        if res < epsilon:

행렬 설계

$$ \begin{bmatrix} \sum 1 & \sum x_i & \cdots & \sum x_i^m \\ \sum x_i & \sum x_i^2& \cdots & \sum x_i^{m+1}\\ \vdots &&\vdots\\ \sum x_i^m & \sum x_i^{m+1}& \cdots & \sum x_i^{2m} \end{bmatrix} \begin{bmatrix} a_0 \\ a_1 \\ \vdots \\ a_m \end{bmatrix} =\begin{bmatrix} \sum f(x_i) \\ \sum f(x_i) x_i \\ \vdots \\ \sum f(x_i)x_i^m \end{bmatrix} $$

이는 아래와 같은 행렬식으로 해석할 수도 있다. 행렬곱이 특이한 연산이라서 가능하다.

$$ \mathbf{X}^T\mathbf{X} \vec{a} = \mathbf{X}^T\vec{y} $$

위 식에서 $\mathbf{X}^T\mathbf{X}$를 $\mathbf{A}$로, $\mathbf{X}^T \vec{y}$를 $\vec{b}$로 치환해서 확인해보면, 가우스-조던 소거법으로 풀이할 수 있음을 알 수 있다.

$$ \mathbf{X}^T \mathbf{X} \vec{a} = \mathbf{X}^T \vec{y} \to \mathbf{A} \vec{a} = \vec{b} $$

$$ \text{where } \mathbf{A} =\mathbf{X}^T \mathbf{X},\ \mathbf{X}^T \vec{y} = \vec{b} $$

즉, 확장행렬을 아래와 같이 구성하고, 가우스-조던 소거법을 적용하면 자연스럽게 $\vec{a}$가 나오고, 그것은 최소제곱법에 의해 추정된 함수의 계수이다.

$$ [\mathbf{X}^T\mathbf{X} | \mathbf{X}^T \vec{y}] \to [I | \vec{a}] \text{ is the soltion of Least Squares.} $$

def GoldenSearch(f, xL, xR, epsilon=1e-10, max_iter=100):
    output = []
    # Corrected golden ratio coefficient
    key_gold = (np.sqrt(5) - 1) / 2
    
    for i in range(max_iter):
        diff = xR - xL
        x1 = xL + (1 - key_gold) * diff
        x2 = xR - (1 - key_gold) * diff

        fx1 = f(x1)
        fx2 = f(x2)

        output.append(f'#iter {i+1}, res = {abs(xR - xL)}, x1 = {x1}, x2 = {x2}, fx1 = {fx1}, fx2 = {fx2}')

        # Stopping condition
        if abs(xR - xL) < epsilon:
            xm = (xL + xR) / 2
            extreme_value = f(xm)
            output.append('\n\n')
            output.append(f'xm = {xm}, extreme value = {extreme_value}')
            for out in output:
                print(out)
            return xm, extreme_value

        # Update the interval based on function evaluations
        if fx1 < fx2:
            xL = x1
        else:
            xR = x2

    # If max iterations reached without convergence
    xm = (xL + xR) / 2
    extreme_value = f(xm)
    output.append('\n\n')
    output.append(f'Max iterations reached: xm = {xm}, extreme value = {extreme_value}')
    for out in output:
        print(out)
    return xm, extreme_value

뉴턴-랩슨법¶

뉴턴-랩슨법을 구현하기 위해선 1계 도함수와 2계 도함수를 알아야 한다.

원래 함수:

$$ g(c) = \frac{2c}{4+0.8c + c^2 + 0.2 c^3} = 2c(4+0.8c + c^2 + 0.2c^3)^{-1} $$

1계 도함수:

$$ f(c) \equiv 2c,\quad g(c) \equiv 4+0.8c + c^2 + 0.2c^3,\quad h(x) \equiv x^{-1} $$

이렇게 치환한 다음 미분의 곱 법칙, 연쇄법칙을 적용하여 풀이한다.

미분의 곱 법칙: $$ (f(c)\cdot h(g(c)))' = f'(c)h(g(c)) + f(c)[h(g(c))]' $$

미분의 연쇄법칙까지 적용: $$ = f'(c)h(g(c)) + f(c)h'(g(c))g'(c) $$

따라서: $$ 2 \cdot (4+0.8c + c^2 + 0.2c^3)^{-1} -2c \cdot (4+0.8c + c^2 + 0.2c^3)^{-2} \cdot (0.8 + 2c + 0.6c^2) $$

다시 분수 표현식으로 바꿔 정리하면:

$$ -\frac{2c(0.8 + 2c + 0.6c^2)}{16(0.05c^3 + c^2/4 + 0.2c + 1)^2} + \frac{2}{4+0.8c + c^2 + 0.2c^3} $$

이것이 1계 도함수이다... 2계 도함수도 어렵게 구해야 할 것이다. 이런 점이 뉴턴-랩슨 방법의 단점이다.

2계 도함수:

첫 번째 항의 도함수: $$ f_1(c) = \frac{c(-0.8 - 2c - 0.6c^2)}{8(0.05c^3 + \frac{c^2}{4} + 0.2c + 1)^2} $$

몫의 법칙 적용: $$ \left( \frac{u}{v} \right)' = \frac{u'v - uv'}{v^2} $$ 여기서:

  • $ u = c(-0.8 - 2c - 0.6c^2) $
  • $ v = 8(0.05c^3 + \frac{c^2}{4} + 0.2c + 1)^2 $

  1. $ u'(c) $ 계산: $$ u = c(-0.8 - 2c - 0.6c^2) $$

    • 곱의 법칙 적용: $$ u' = 1 \cdot (-0.8 - 2c - 0.6c^2) + c \cdot (-2 - 1.2c) $$
      • 결과: $$ u' = -0.8 - 2c - 0.6c^2 - 2c - 1.2c^2 = -0.8 - 4c - 1.8c^2 $$

  2. $ v'(c) $ 계산: $$ v = 8(0.05c^3 + \frac{c^2}{4} + 0.2c + 1)^2 $$

    • 연쇄 법칙 및 곱의 법칙 적용: $$ v' = 8 \cdot 2(0.05c^3 + \frac{c^2}{4} + 0.2c + 1) \cdot (0.15c^2 + \frac{c}{2} + 0.2) $$
      • 결과: $$ v' = 16(0.05c^3 + \frac{c^2}{4} + 0.2c + 1)(0.15c^2 + \frac{c}{2} + 0.2) $$

  3. $ f_1'(c) $: $$ f_1'(c) = \frac{u'v - uv'}{v^2} $$

두 번째 항의 도함수 $$ f_2(c) = \frac{2}{4 + 0.8c + c^2 + 0.2c^3} $$

몫의 법칙 적용: 여기서:

  • $ u = 2 $
  • $ v = 4 + 0.8c + c^2 + 0.2c^3 $

  1. $ u'(c) = 0 $ (상수항의 도함수)

  2. $ v'(c) $: $$ v' = 0.8 + 2c + 0.6c^2 $$

  3. $ f_2'(c) $: $$ f_2'(c) = \frac{0 \cdot v - 2 \cdot v'}{v^2} = \frac{-2(0.8 + 2c + 0.6c^2)}{(4 + 0.8c + c^2 + 0.2c^3)^2} $$

전체 도함수 전체 도함수는 두 항의 도함수를 합친 결과: $$ f'(c) = f_1'(c) + f_2'(c) $$

정리:

  1. $ f_1'(c) $: $$ f_1'(c) = \frac{(-0.8 - 4c - 1.8c^2)(8(0.05c^3 + \frac{c^2}{4} + 0.2c + 1)^2) - (c(-0.8 - 2c - 0.6c^2))(16(0.05c^3 + \frac{c^2}{4} + 0.2c + 1)(0.15c^2 + \frac{c}{2} + 0.2))}{[8(0.05c^3 + \frac{c^2}{4} + 0.2c + 1)^2]^2} $$

  2. $ f_2'(c) $: $$ f_2'(c) = \frac{-2(0.8 + 2c + 0.6c^2)}{(4 + 0.8c + c^2 + 0.2c^3)^2} $$

import numpy as np
import sympy as sp

def newton_raphson(func:sp.Expr, var:sp.Symbol, initial_guess:float, tolerance:float = 1e-4, max_iterations:int = 100):
    """
    # Newton-Raphson method to find the maximum of a single-variable function.
    
    ## Parameters:
        func (sympy expression): The function to maximize.
        var (sympy Symbol): The variable in the function.
        initial_guess (float): The starting point for the iteration.
        tolerance (float): Convergence criterion for the iteration.
        max_iterations (int): Maximum number of iterations.
    
    Returns:
        output_tuple:
        max_point: The x-coordinate of the maximum point.

        max_value: The maximum value of the function.

        iterations: The number of iterations performed.
    """
    # Compute the first and second derivatives of the function
    first_derivative = sp.diff(func, var)
    second_derivative = sp.diff(first_derivative, var)
    
    # Initialize variables
    x = initial_guess
    iterations = 0
    xhistory = []
    yhistory = []
    xhistory.append(x)
    yhistory.append(func.subs(var, x))

    for _ in range(max_iterations):
        iterations += 1
        
        # Evaluate the derivatives at the current point
        f_prime = first_derivative.subs(var, x)
        f_double_prime = second_derivative.subs(var, x)
        
        # Newton-Raphson update
        if f_double_prime == 0:  # Avoid division by zero
            raise ValueError("Second derivative is zero. Cannot continue.")
        
        x_new = x - f_prime / f_double_prime
        
        # Check for convergence
        if abs(x_new - x) < tolerance:
            x = x_new
            break
        
        x = x_new
        xhistory.append(x)
        yhistory.append(func.subs(var, x))
    
    max_point = x
    max_value = func.subs(var, max_point)
    
    return float(max_point), float(max_value), iterations, xhistory, yhistory

Hornbeck 9.18¶

해석해 구하기¶

$ \frac{dy}{dt} + y^2 = 1 $ 분리 변수 형태로 정리:

$$ \frac{dy}{1 - y^2} = dt $$

1단계: 부분 분수 분해

분모 $ 1 - y^2 $을 다음과 같이 분해: $$ 1 - y^2 = (1 - y)(1 + y) $$

따라서: $$ \frac{1}{1 - y^2} = \frac{A}{1 - y} + \frac{B}{1 + y} $$

여기서 $ A $와 $ B $를 구하기 위해 부분 분수 분해를 적용: $$ 1 = A(1 + y) + B(1 - y) $$

$ y = 1 $을 대입하면: $ 1 = 2A \implies A = \frac{1}{2} $
$ y = -1 $을 대입하면: $ 1 = 2B \implies B = \frac{1}{2} $

따라서: $$ \frac{1}{1 - y^2} = \frac{1/2}{1 - y} + \frac{1/2}{1 + y} $$

2단계: 적분하기

위 식을 적분: $$ \int \frac{1}{1 - y^2} \, dy = \int \left( \frac{1/2}{1 - y} + \frac{1/2}{1 + y} \right) dy $$

각 항을 적분하면: $$ \int \frac{1/2}{1 - y} \, dy = -\frac{1}{2} \ln |1 - y|, \quad \int \frac{1/2}{1 + y} \, dy = \frac{1}{2} \ln |1 + y| $$

따라서: $$ \int \frac{1}{1 - y^2} \, dy = -\frac{1}{2} \ln |1 - y| + \frac{1}{2} \ln |1 + y| = \frac{1}{2} \ln \left| \frac{1 + y}{1 - y} \right| $$

이 식을 미분방정식에 대입: $$ \frac{1}{2} \ln \left| \frac{1 + y}{1 - y} \right| = t + C $$

3단계: $ y $에 대해 풀기

양변에 지수를 취한다: $$ \ln \left| \frac{1 + y}{1 - y} \right| = 2t + 2C $$

$ K = e^{2C} $ (상수)로 두면: $$ \frac{1 + y}{1 - y} = K e^{2t} $$

$ y $를 풀이: $$ 1 + y = (1 - y) K e^{2t} $$

$$ 1 + y = K e^{2t} - K e^{2t} y $$

$$ y (1 + K e^{2t}) = K e^{2t} - 1 $$

$$ y = \frac{K e^{2t} - 1}{1 + K e^{2t}} $$

4단계: 초기 조건으로 $ K $ 구하기

초기 조건 $ t = 0 $에서 $ y(0) = 0 $을 대입: $$ 0 = \frac{K e^{0} - 1}{1 + K e^{0}} $$

$$ 0 = \frac{K - 1}{1 + K} $$

$$ K = 1 $$

따라서: $$ y = \frac{e^{2t} - 1}{e^{2t} + 1} $$

최종 해석해:

$$ \therefore\ y(t) = \frac{e^{2t} - 1}{e^{2t} + 1} $$

def RK4(f, y0: float, t0: float, t_end: float, n: int) -> tuple:
    """
    Implements the fourth-order Runge-Kutta (RK4) method to numerically solve 
    ordinary differential equations (ODEs) of the form dy/dt = f(y, t).

    Parameters:
        f (callable): The function f(y, t) that defines the ODE. 
                      It should take two arguments (y, t) and return the derivative dy/dt.
        y0 (float): Initial value of the dependent variable y at t = t0.
        t0 (float): Initial time value.
        t_end (float): End time value up to which the ODE is solved.
        n (int): Number of time steps (including t0 and t_end).

    Returns:
        tuple: A tuple containing:
            - t_values (numpy.ndarray): Array of time values from t0 to t_end with n points.
            - y_values (numpy.ndarray): Array of y values corresponding to each time value.

    Example:
        >>> import numpy as np
        >>> def f(y, t):
        >>>     return 1 - y**2
        >>> t_values, y_values = RK4(f, y0=0, t0=0, t_end=2.5, n=100)
        >>> print(t_values)
        >>> print(y_values)
    """
    y = np.array(y0, dtype=np.float64)
    t_values = np.linspace(t0, t_end, num=n, endpoint=True, dtype=np.float64)
    delt = (t_end - t0)/n
    y_values = [y0]

    for t in t_values[:-1]:
        y_halfj = y + (delt/2)*f(y, t)
        y_halfj_star = y + (delt/2)*f(y_halfj, t + (delt/2))
        y_next_star = y + delt*f(y_halfj_star, t + (delt/2))
        y = y + delt*((1/6)*f(y, t) + (1/3)*f(y_halfj, t) + (1/3)*f(y_halfj_star, t) + (1/6)*f(y_next_star, t))
        y_values.append(y)

    return t_values, np.array(y_values, dtype=np.float64)


def RK4_2nd(dfz, dfy, y0, z0, t0, t_end, n):
    """
    # Quad-order Runge-Kutta (RK4) method for solving Second-order ODEs.

    ## Parameters:
    - **dfz**: Function dz/dt = f(y, z, t)
    - **dfy**: FUnction dy/dt = z = f(y, z, t)
    - **y0**: Initial value y(t0)
    - **z0**: Initial value z(t0) = dy(t0)/dt
    - **t0**: Initial time
    - **t_end**: End time
    - **n**: Number of points

    Returns:
    - **t_values**: Time points
    - **z_values**: Solution at each time point for z
    - **y_values**: Solution at each time point for y
    """
    t_values = np.linspace(t0, t_end, n, endpoint=True)
    y_values = [y0]
    z_values = [z0]
    delt = (t_end - t0) / (n)  # Step size

    y = y0
    z = z0
    for t in t_values[:-1]:
        z_halfj = z + (delt/2)*dfz(y, z, t)
        y_halfj = y + (delt/2)*dfy(y, z, t)
        z_halfj_star = z + (delt/2)*dfz(y_halfj, z_halfj, t + (delt/2))
        y_halfj_star = y + (delt/2)*dfy(y_halfj, z_halfj, t + (delt/2))
        z_next_star = z + delt*dfz(y_halfj_star, z_halfj_star, t + (delt/2))
        y_next_star = y + delt*dfy(y_halfj_star, z_halfj_star, t + (delt/2))
        z = z + delt*((1/6)*dfz(y, z, t) + (1/3)*dfz(y_halfj, z_halfj, t) + (1/3)*dfz(y_halfj_star, z_halfj_star, t) + (1/6)*dfz(y_next_star, z_next_star, t))

        z_values.append(z)
        y_values.append(y)

    return t_values, np.array(z_values), np.array(y_values)
