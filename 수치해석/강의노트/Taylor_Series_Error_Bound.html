import math
import matplotlib.pyplot as plt
import numpy as np

x_values = np.linspace(-2 * np.pi, 2 * np.pi, 400)

#    e^x  
def taylor_exp(x, n):
    approx = 0
    for i in range(n + 1):
        term = (x**i) / math.factorial(i)  # (x^i) / i!
        approx += term
    return approx

#  e^x     
exp_values = np.exp(x_values)

#     (1, 2, 3)
taylor_exp_1_values = [taylor_exp(x, 1) for x in x_values]  # 1
taylor_exp_2_values = [taylor_exp(x, 2) for x in x_values]  # 2
taylor_exp_3_values = [taylor_exp(x, 3) for x in x_values]  # 3

#  
plt.figure(figsize=(10, 6))
plt.plot(x_values, exp_values, label='e^x', color='blue')
plt.plot(x_values, taylor_exp_1_values, label='Taylor Approx (1st term)', linestyle='--', color='red')
plt.plot(x_values, taylor_exp_2_values, label='Taylor Approx (2nd term)', linestyle='--', color='green')
plt.plot(x_values, taylor_exp_3_values, label='Taylor Approx (3rd term)', linestyle='--', color='orange')

#  
plt.title('Taylor Series Approximation of e^x', fontsize=16)
plt.xlabel('x', fontsize=14)
plt.ylabel('e^x', fontsize=14)
plt.legend()
plt.grid(True)

#  
plt.show()

import math
import matplotlib.pyplot as plt
import numpy as np

#    sin(x)  
def taylor_sin(x, n):
    approx = 0
    for i in range(n + 1):
        coeff = (-1)**i  #  (-1)^i
        term = (x**(2 * i + 1)) / math.factorial(2 * i + 1)  # (x^(2i+1)) / (2i+1)!
        approx += coeff * term
    return approx

#  sin(x)     
x_values = np.linspace(-2 * np.pi, 2 * np.pi, 400)
sin_values = np.sin(x_values)

#     (3, 5, 7)
taylor_3_values = [taylor_sin(x, 1) for x in x_values]  # 3 (1)
taylor_5_values = [taylor_sin(x, 2) for x in x_values]  # 5 (2)
taylor_7_values = [taylor_sin(x, 3) for x in x_values]  # 7 (3)

#  
plt.figure(figsize=(10, 6))
plt.plot(x_values, sin_values, label='sin(x)', color='blue')
plt.plot(x_values, taylor_3_values, label='Taylor Approx (1st term)', linestyle='--', color='red')
plt.plot(x_values, taylor_5_values, label='Taylor Approx (2nd term)', linestyle='--', color='green')
plt.plot(x_values, taylor_7_values, label='Taylor Approx (3rd term)', linestyle='--', color='orange')

#  
plt.title('Taylor Series Approximation of sin(x)', fontsize=16)
plt.xlabel('x', fontsize=14)
plt.ylabel('sin(x)', fontsize=14)
plt.legend()
plt.grid(True)

import math

#     
def truncation_error_exp(x, n):
    #    for e^x
    error = abs((math.exp(x)) / math.factorial(n + 1)) * abs(x ** (n + 1))
    return error

def truncation_error_sin(x, n):
    #    for sin(x), sin   1
    error = abs(1 / math.factorial(n + 1)) * abs(x ** (n + 1))
    return error

x_sin = 1  # sin(x)  x
n_sin = 1  # sin(x) 1

# sin(x)  
sin_error = truncation_error_sin(x_sin, n_sin)

print(f"e^{x_exp}  (n = {n_exp}): {exp_error}")
